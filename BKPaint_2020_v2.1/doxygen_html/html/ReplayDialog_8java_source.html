87  public void saveFile() {
88  JFileChooser fileSave = new JFileChooser("Save a replay file");
89  int select = 0;
90  File init = new File("Untitled.rep");
91  fileSave.setSelectedFile(init);
92  select = fileSave.showSaveDialog(null); //Hien thi filechoser cung voi ten mac dinh
93  if (select == JFileChooser.APPROVE_OPTION) {
94  File file = fileSave.getCurrentDirectory();
95  String fileName = file.getPath() + "\\" + fileSave.getSelectedFile().getName();
96  file = new File(fileName);
97  if (file.exists()) {
98  int r = JOptionPane.showConfirmDialog(this, "File" + fileSave.getSelectedFile() + " already exists\nDo you want to replace it", "Warning", JOptionPane.YES_NO_OPTION, JOptionPane.WARNING_MESSAGE);
99  if (r == JOptionPane.NO_OPTION) {
100  return;
101  }
102  }
103  try {
104  ObjectOutputStream oos = new ObjectOutputStream(new FileOutputStream(fileName));
105  oos.writeObject(paintState);
106  saveToFile = true;
107  oos.close();
108  } catch (IOException ex) {
109  JOptionPane.showMessageDialog(null, "Save file error!", "Error", JOptionPane.ERROR_MESSAGE);
110  Logger.getLogger(ReplayDialog.class.getName()).log(Level.SEVERE, null, ex);
111  }
112  }
113  }
